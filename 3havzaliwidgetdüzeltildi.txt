        function updateDashState(index) {
            if(!dashEChart || index >= fullDates.length) return;
            const dateStr = fullDates[index]; 
            const twsaVal = mergedTWSAData[index];
            const currentYear = parseInt(dateStr.substring(0, 4));
            
            let badgeHtml = currentYear >= 2026 ? '<span class="badge badge-ai">AI TAHMİNİ</span>' : '<span class="badge badge-grace">GRACE UYDUSU</span>';
            let textColor = '#00BFFF'; if (twsaVal < 0) textColor = '#FFA500'; if (twsaVal < -10) textColor = '#ff4444';

            document.getElementById('dash-date').innerHTML = `${dateStr} ${badgeHtml}`;
            const twsaDisplay = document.getElementById('dash-twsa');
            twsaDisplay.innerText = `TWSA: ${twsaVal > 0 ? '+' : ''}${twsaVal} cm`;
            twsaDisplay.style.color = textColor;

            let slicedHist = fullHistData.map((val, i) => i <= index ? val : '-');
            let slicedPred = fullPredData.map((val, i) => i <= index ? val : '-');

            dashEChart.setOption({ series: [ { data: slicedHist }, { data: slicedPred } ] });

            if (map.getLayer('havzalar-fill')) {
                let mapColor = 'rgba(135, 206, 235, 0.4)'; 
                if(twsaVal < 0) mapColor = 'rgba(255, 165, 0, 0.5)';

            }

        function processUploadedData() {
            const fileInput = document.getElementById('custom-data-upload');
            if (!fileInput.files.length) { Swal.fire({ icon: 'error', title: 'Hata', text: 'Lütfen bir CSV dosyası seçin!', background: '#14161c', color: '#fff' }); return; }
            const file = fileInput.files[0]; const reader = new FileReader();
            reader.onload = function(e) {
                const text = e.target.result; const rawValues = text.split(/[\n,;]+/).map(v => parseFloat(v.trim())).filter(v => !isNaN(v));
                if (rawValues.length < 50) { Swal.fire({ icon: 'warning', title: 'Yetersiz Veri', text: 'CSV dosyasında yeterli veri (aylık tahmin) bulunamadı.', background: '#14161c', color: '#fff' }); return; }
                mergedTWSAData = []; fullHistData = []; fullPredData = [];
                for (let i = 0; i < fullDates.length; i++) {
                    let val = rawValues[i] !== undefined ? rawValues[i].toFixed(2) : (rawValues[rawValues.length-1] - (i * 0.05)).toFixed(2); mergedTWSAData.push(parseFloat(val));
                    if (fullDates[i] < "2026-01") { fullHistData.push(val); fullPredData.push('-'); } else if (fullDates[i] === "2026-01") { fullHistData.push(val); fullPredData.push(val); } else { fullHistData.push('-'); fullPredData.push(val); }
                }
                if(dashEChart) initDashChart(); if(activePanelChart) initPanelECharts('echarts-container'); updateDashState(0); document.getElementById('dash-slider').value = 0; scrollToSection('dashboard-section');
                Swal.fire({ icon: 'success', title: 'Sistem Güncellendi!', text: 'Yeni AI modeliniz Karar Destek Sistemine başarıyla entegre edildi.', background: '#14161c', color: '#fff', confirmButtonColor: '#00BFFF' });
            }; reader.readAsText(file);
        }
